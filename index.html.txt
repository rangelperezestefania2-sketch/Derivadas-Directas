function calcularDerivadaPolinomio() {
    // Tomamos la ecuación y le quitamos los espacios para que no haya errores
    let texto = document.getElementById('funcionInput').value.replace(/\s+/g, '');
    if (!texto) return;

    // Con este truco separamos la función en partes usando los signos + y -
    let terminos = texto.match(/([+-]?[^+-]+)/g);
    
    let pasosHTML = `<p>Dada la función: $$f(x) = ${texto}$$</p>`;
    pasosHTML += `<p>Derivamos cada término por separado usando las reglas básicas:</p>`;
    pasosHTML += `<ol style="margin-left: 20px; margin-top: 10px;">`;
    
    let resultadoFinalPartes = [];

    // Revisamos cada pedacito por separado
    terminos.forEach((term, index) => {
        // Guardamos el signo que tiene el término
        let signoOriginal = term.startsWith('-') ? '-' : (index > 0 ? '+' : '');
        let limpio = term.replace(/^[+-]/, ''); // Le quitamos el signo del inicio para trabajar limpio

        // CASO 1: Términos con potencia como x^2 o 3x^4
        if (limpio.includes('x^')) {
            let partes = limpio.split('x^');
            let c = partes[0] === '' ? 1 : parseFloat(partes[0]); // Si no hay número antes de x, es un 1
            let n = parseFloat(partes[1]);
            
            let nuevoC = c * n;
            let nuevoN = n - 1;
            
            // Si el nuevo exponente es 1, no se escribe el "^1"
            let textoExponente = nuevoN === 1 ? 'x' : `x^{${nuevoN}}`;
            let terminoDerivado = `${signoOriginal}${nuevoC}${textoExponente}`;
            
            pasosHTML += `<li>Para $$${term}$$: Multiplicamos el exponente por el coeficiente ($$${n} \\cdot ${c} = ${nuevoC}$$) y restamos 1 al exponente ($$${n} - 1 = ${nuevoN}$$) ➔ $$${terminoDerivado}$$</li>`;
            resultadoFinalPartes.push(terminoDerivado);
        } 
        // CASO 2: Términos con x lineal como 10x
        else if (limpio.includes('x')) {
            let c = limpio.replace('x', '') === '' ? 1 : parseFloat(limpio.replace('x', ''));
            let terminoDerivado = `${signoOriginal}${c}`;
            
            pasosHTML += `<li>Para $$${term}$$: La derivada de $$x$$ es 1, por lo tanto solo se conserva su coeficiente constante ➔ $$${terminoDerivado}$$</li>`;
            resultadoFinalPartes.push(terminoDerivado);
        } 
        // CASO 3: Números solos como el 5
        else {
            pasosHTML += `<li>Para el número solo $$${term}$$: La derivada de cualquier constante numérica es siempre 0.</li>`;
        }
    });

    pasosHTML += `</ol>`;
    
    // Juntamos todas las respuestas individuales para armar la solución final
    let resultadoUnido = resultadoFinalPartes.join('');
    // Corrección por si se juntan signos raros como +-
    resultadoUnido = resultadoUnido.replace(/\+\-/g, '-').replace(/^\+/, '');

    pasosHTML += `<br><strong>Resultado Final:</strong> $$f'(x) = ${resultadoUnido}$$`;

    // Ponemos todo en la pantalla del celular y le decimos a MathJax que lo dibuje bonito
    document.getElementById('pasosMath').innerHTML = pasosHTML;
    document.getElementById('resultadoArea').style.display = 'block';
    MathJax.typesetClear();
    MathJax.typeset();
}
